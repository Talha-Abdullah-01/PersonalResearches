// maximum is a global variable.
// Updates maximum if parameter is larger
void set_max(int a)
{
    static int maximum;
    if (maximum < a)
        maximum = a;
    cout<< maximum;
}
int max3(int i, int j, int k)
{
    static int maximum;
    maximum = i;
    set_max(j);
    set_max(k);
    return maximum;
}
